## Attributes

<ResponseField name="id" type="string">
Unique identifier for the client session.

Example: `"d3b07384-d9a5-4d16-a5b1-3fa3d9b0b123"`
</ResponseField>

<ResponseField name="expiry_date" type="string">
Expiration date and time of the client session.

Example: `"2021-08-12T16:14:08.578695"`
</ResponseField>

<ResponseField name="customer" type="object">
Customer information associated with the client session.

<Expandable title="Customer properties">
<ResponseField name="address" type="object">
Address information of the customer.

<Expandable title="Address properties">
<ResponseField name="country" type="string">
ISO country code (e.g., US, DE, AD).
</ResponseField>
</Expandable>
</ResponseField>

<ResponseField name="email" type="string">
Email address of the customer.

Example: `"jane.doe@example.com"`
</ResponseField>

<ResponseField name="external_id" type="string">
External ID of the customer.

Example: `"cust_123456789"`
</ResponseField>

<ResponseField name="metadata" type="object">
Custom metadata. A dictionary of key-value pairs.

<Expandable title="Metadata properties">
<ResponseField name="{key}" type="string">
Custom key-value pairs for storing additional data.
</ResponseField>
</Expandable>
</ResponseField>
</Expandable>
</ResponseField>

<ResponseField name="plan" type="object">
Subscription plan information.

<Expandable title="Plan properties">
    <ResponseField name="id" type="string">
    Unique identifier for the plan.
    
    Example: `"1a2b3c4d-5e6f-7a8b-9c0d-1e2f3a4b5c6d"`
    </ResponseField>

    <ResponseField name="created_at" type="string">
    Timestamp when the plan was created.
    
    Example: `"2025-05-26T10:00:00Z"`
    </ResponseField>

    <ResponseField name="updated_at" type="string">
    Timestamp when the plan was last updated.
    
    Example: `"2025-05-27T10:00:00Z"`
    </ResponseField>

    <ResponseField name="name" type="string">
    Name of the subscription plan.
    
    Example: `"Pro Plan"`
    </ResponseField>

    <ResponseField name="type" type="string">
    Type of the subscription plan.
    
    Example: `"recurring"`
    </ResponseField>

    <ResponseField name="interval" type="string">
    Billing interval unit.
    
    Example: `"days"`
    </ResponseField>

    <ResponseField name="interval_count" type="integer">
    Number of intervals between billing.
    
    Example: `1`
    </ResponseField>

    <ResponseField name="amount" type="integer">
    Plan amount in the smallest currency unit.
    
    Example: `1999`
    </ResponseField>

    <ResponseField name="currency" type="string">
    Currency code for the plan.
    
    Example: `"AED"`
    </ResponseField>

    <ResponseField name="trial" type="object">
    Trial period information.
    
    <Expandable title="Trial properties">
        <ResponseField name="interval" type="string">
        Trial interval unit.
        
        Example: `"days"`
        </ResponseField>

        <ResponseField name="interval_count" type="integer">
        Number of trial intervals.
        
        Example: `1`
        </ResponseField>

        <ResponseField name="amount" type="integer">
        Trial amount in the smallest currency unit.
        
        Example: `1999`
        </ResponseField>

        <ResponseField name="currency" type="string">
        Trial currency code.
        
        Example: `"AED"`
        </ResponseField>
    </Expandable>
    </ResponseField>
</Expandable>
</ResponseField>

<ResponseField name="currency_details" type="object">
Detailed currency information for the plan.

<Expandable title="Currency details properties">

            <ResponseField name="currency_decimals" type="integer">

            Example: `2`
            </ResponseField>

            <ResponseField name="currency_exchange_rate" type="number">
            Exchange rate from payment currency to local currency.
            
            Example: `0.9`
            </ResponseField>

            <ResponseField name="currency_position" type="string">
            Position of currency symbol.
            
            Example: `"left"`
            </ResponseField>

            <ResponseField name="currency_symbol" type="string">
            Currency symbol.
            
            Example: `"€"`
            </ResponseField>

            <ResponseField name="formatted_amount" type="string">
            Formatted amount with currency symbol.

            <ResponseField name="formatted_amount_rounded" type="string">
            Rounded formatted amount with currency symbol.
            
            Example: `"€0.99"`
