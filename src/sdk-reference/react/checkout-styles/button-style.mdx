---
title: 'Button Style'
---

<ResponseField name="button" type="object">
  An object that defines the style of the input element.

  <ResponseField name="styles" type="CSSProperties">
    CSS properties to apply to the input element.
  </ResponseField>

  <ResponseField name="className" type="string">
    Class name to apply to the input element.
  </ResponseField>

  <ResponseField name="icon" type="ReactNode">
    Icon which can applied to the submit button.
  </ResponseField>
</ResponseField>
